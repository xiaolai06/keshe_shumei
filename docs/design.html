项目概述

Project Overview

本项目以 Raspberry Pi 为硬件实体，构建一个具备多模态感知、三层记忆系统、提醒规划能力的 AI 桌宠智能体。桌宠通过 OLED 展示表情与文字、RGB LED 表达情绪色彩、蜂鸣器/USB 音响发出提示音，同时通过麦克风接收语音、摄像头捕获画面，传感器感知环境。提供 Web 网页端实现远程对话、数据看板、摄像头画面实时查看和记忆回顾。

# 方案2: 直接播放 WAV
import pyaudio, wave

def play_wav(filepath):
    wf = wave.open(filepath, 'rb')
    p = pyaudio.PyAudio()
    stream = p.open(format=p.get_format_from_width(wf.getsampwidth()),
                    channels=wf.getnchannels(),
                    rate=wf.getframerate(),
                    output=True)
    data = wf.readframes(1024)
    while data:
        stream.write(data)
        data = wf.readframes(1024)
    stream.close()
    p.terminate()

记忆系统

Three-Layer Memory Architecture

三层架构

┌──────────────────────────────────────────────────────────────┐ │ 记忆系统 │ │ │ │ ┌──────────────────────────────────────────────────────┐ │ │ │ Layer 1: 工作记忆 (Working Memory) │ │ │ │ 最近 10 轮对话 + 当前情绪 + 传感器快照 │ │ │ │ 存储: LangGraph State (内存) │ │ │ └──────────────────────────────────────────────────────┘ │ │ │ 压缩 │ │ ▼ │ │ ┌──────────────────────────────────────────────────────┐ │ │ │ Layer 2: 短期记忆 (Short-term) │ │ │ │ 今日交互摘要 + 近期传感器趋势 │ │ │ │ 存储: SQLite short_term_memory 保留 7 天 │ │ │ └──────────────────────────────────────────────────────┘ │ │ │ 精选 │ │ ▼ │ │ ┌──────────────────────────────────────────────────────┐ │ │ │ Layer 3: 长期记忆 (Long-term) │ │ │ │ 重要事件 + 用户偏好 + 语义记忆 │ │ │ │ 存储: SQLite (元数据) + ChromaDB (向量检索) │ │ │ └──────────────────────────────────────────────────────┘ │ └──────────────────────────────────────────────────────────────┘

操作流程

  • 1

    写入 — 每次交互

    将对话、传感器快照、情绪写入短期记忆表

  • 2

    检索 — Recall 节点

    工作记忆取上下文；ChromaDB 语义检索长期记忆

  • 3

    总结 — Reflect 节点（定时）

    每 N 次交互，LLM 总结短期记忆 → 写入长期记忆

  • 4

    遗忘 — 过期清理

    短期记忆 7 天后自动归档，长期记忆降低低频权重

class MemoryManager:
    """融合 MemGPT 分层 + 斯坦福评分检索"""

    def store(self, event: str, importance: int = None):
        """
        存入一条记忆
        - importance 为 None 时自动让 LLM 评分
        """
        if importance is None:
            importance = self.llm_rate_importance(event)  # 1-10

        embedding = self.embed(event)
        self.db.execute(
            "INSERT INTO memories (content, importance, embedding_id, created_at) "
            "VALUES (?, ?, ?, ?)",
            (event, importance, self.chroma.add(embedding), datetime.now())
        )

    def recall(self, query: str, top_k: int = 5) -> list:
        """
        检索最相关记忆 (三维度打分)
        """
        candidates = self.chroma.query(self.embed(query), n=top_k * 3)
        scored = []
        for c in candidates:
            recency = math.exp(-(now - c.created_at).total_seconds() / 86400)
            relevance = cosine_similarity(self.embed(query), c.embedding)
            importance = c.importance / 10
            score = recency * 0.3 + relevance * 0.5 + importance * 0.2
            scored.append((score, c))
        return sorted(scored, reverse=True)[:top_k]

    def reflect(self, recent_memories: list):
        """
        反思：当高重要性记忆积累足够时触发
        """
        summary = self.llm_summarize(recent_memories)
        self.store(summary, importance=8)  # 反思结果高重要性

🌡️ 环境舒适度因子 (env_comfort)

基于 DS18B20 温度和 DHT11 湿度计算环境舒适度，直接影响 happy/sleepy 得分。

def calc_env_comfort(temp: float, humidity: float) -> float:
    """环境舒适度 0~1，1=非常舒适"""
    # 温度舒适区间: 20~26°C
    if 20 <= temp <= 26:
        temp_score = 1.0
    elif temp < 10 or temp > 38:
        temp_score = 0.0
    elif temp < 20:
        temp_score = (temp - 10) / 10  # 线性插值
    else:
        temp_score = (38 - temp) / 12

    # 湿度舒适区间: 40%~70%
    if 40 <= humidity <= 70:
        humi_score = 1.0
    elif humidity < 20 or humidity > 90:
        humi_score = 0.0
    elif humidity < 40:
        humi_score = (humidity - 20) / 20
    else:
        humi_score = (90 - humidity) / 20

    return temp_score * 0.6 + humi_score * 0.4  # 温度权重更高

def calc_interaction_sentiment(recent_messages: list) -> dict:
    """
    对最近对话做情感分析
    返回: {positive: 0~1, negative: 0~1, question_rate: 0~1, msg_rate: msgs/min}
    """
    if not recent_messages:
        return {"positive": 0.5, "negative": 0.0, "question_rate": 0, "msg_rate": 0}

    # 方案 A: 简单关键词匹配 (无需额外 API 调用)
    pos_words = {"谢谢","开心","好的","太棒","喜欢","厉害","不错","哈哈","可爱"}
    neg_words = {"不好","难过","无聊","生气","讨厌","烦","糟糕","累"}

    pos = sum(1 for m in recent_messages if any(w in m for w in pos_words))
    neg = sum(1 for m in recent_messages if any(w in m for w in neg_words))
    questions = sum(1 for m in recent_messages if "？" in m or "?" in m)
    total = len(recent_messages)

    # 方案 B: 让 LLM 做一次批量情感分析 (每 10 条调一次，节省 token)
    # sentiment = llm_analyze_sentiment(recent_messages)

    time_span = (recent_messages[-1].time - recent_messages[0].time).seconds / 60
    return {
        "positive": pos / total,
        "negative": neg / total,
        "question_rate": questions / total,
        "msg_rate": total / max(time_span, 1)
    }

from datetime import datetime

def calc_time_factor(days_alive: int) -> dict:
    """时间因子: 影响 sleepy/happy/lonely"""
    now = datetime.now()
    hour = now.hour

    # 时段因子
    if 6 <= hour < 10:
        period = "morning"    # 早上: 精力恢复
        sleepy_factor = 0.2
        energy_bonus = 0.7
    elif 10 <= hour < 14:
        period = "noon"       # 中午: 正常
        sleepy_factor = 0.3
        energy_bonus = 0.8
    elif 14 <= hour < 18:
        period = "afternoon"  # 下午: 正常偏活跃
        sleepy_factor = 0.3
        energy_bonus = 0.7
    elif 18 <= hour < 22:
        period = "evening"    # 晚上: 活跃时段
        sleepy_factor = 0.2
        energy_bonus = 0.9
    else:
        period = "night"      # 深夜: 困倦
        sleepy_factor = 0.8
        energy_bonus = 0.2

    # 存活天数加成 (活得越久越沉稳)
    wisdom_bonus = min(days_alive / 100, 1.0) * 0.3

    return {
        "period": period,
        "sleepy_factor": sleepy_factor,
        "energy_bonus": energy_bonus,
        "wisdom_bonus": wisdom_bonus,
    }

检测传感器数据是否异常，用于触发 alert 情绪。

def calc_anomaly(current: dict, history_avg: dict) -> float:
    """
    异常分数 0~1
    基于当前值与近期均值的偏离程度
    """
    temp_delta = abs(current["temp"] - history_avg["temp"])
    humi_delta = abs(current["humidity"] - history_avg["humidity"])

    # 温度偏离 > 5°C 或湿度偏离 > 15% 视为异常
    temp_anomaly = min(temp_delta / 5, 1.0)
    humi_anomaly = min(humi_delta / 15, 1.0)

    return max(temp_anomaly, humi_anomaly)

async function refreshChart(range = '24h') {
    const res = await fetch(`/api/sensors/history?range=${range}`);

function drawMoodTimeline(canvasId, moodLog) {
    const canvas = document.getElementById(canvasId);
    const ctx = canvas.getContext('2d');
    const width = canvas.width;
    const segWidth = width / moodLog.length;
    moodLog.forEach((entry, i) => {

        ctx.fillRect(i * segWidth, 0, segWidth, 40);
    });
}

后端数据聚合

# 数据聚合: 10秒原始数据太多，按时间范围做降采样
@app.get("/api/sensors/history")
async def sensor_history(range: str = "24h"):
    """返回降采样后的传感器数据"""
    intervals = {"1h": 10, "6h": 60, "24h": 300, "7d": 1800}  # 采样间隔(秒)
    step = intervals.get(range, 300)

    rows = db.execute(f"""
        SELECT
            timestamp,
            ROUND(AVG(temperature), 1) as temperature,
            ROUND(AVG(humidity), 1) as humidity
        FROM sensor_readings
        WHERE timestamp > datetime('now', ?)
        GROUP BY CAST(strftime('%s', timestamp) / {step} AS INTEGER)
        ORDER BY timestamp
    """, [f"-{range}"]).fetchall()

    return {"success": True, "data": rows}

-- 传感器数据
CREATE TABLE sensor_readings (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    timestamp   DATETIME DEFAULT CURRENT_TIMESTAMP,
    temperature REAL,
    humidity    REAL,
    light_level REAL
);

-- 交互记录
CREATE TABLE interactions (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    timestamp   DATETIME DEFAULT CURRENT_TIMESTAMP,
    source      TEXT,          -- 'voice'/'web'/'system'
    user_input  TEXT,
    agent_reply TEXT,
    mood        TEXT,
    tools_used  TEXT           -- JSON array
);

-- 长期记忆
CREATE TABLE long_term_memory (
    id           INTEGER PRIMARY KEY AUTOINCREMENT,
    category     TEXT,          -- 'event'/'preference'/'reflection'
    content      TEXT,
    importance   REAL DEFAULT 5,
    embedding_id TEXT,          -- ChromaDB 向量 ID
    created_at   DATETIME DEFAULT CURRENT_TIMESTAMP,
    last_recalled DATETIME
);

-- 提醒
CREATE TABLE reminders (
    id         INTEGER PRIMARY KEY AUTOINCREMENT,
    content    TEXT NOT NULL,
    type       TEXT DEFAULT 'once',
    trigger_at DATETIME,
    cron_expr  TEXT,
    condition  TEXT,            -- JSON
    is_active  BOOLEAN DEFAULT 1,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 情绪日志 (用于网页端情绪时间线图表)
CREATE TABLE mood_log (
    id         INTEGER PRIMARY KEY AUTOINCREMENT,
    timestamp  DATETIME DEFAULT CURRENT_TIMESTAMP,
    mood       TEXT NOT NULL,   -- happy/curious/sleepy/alert/chatty/calm/lonely
    score      REAL,            -- 该情绪的得分 0~1
    factors    TEXT             -- JSON: {env: 0.8, interact: 0.6, time: 0.3, ...}
);

-- 进化状态
CREATE TABLE evolution_state (
    id       INTEGER PRIMARY KEY DEFAULT 1,
    name     TEXT DEFAULT '小派',
    mood     TEXT DEFAULT 'happy',
    level    INTEGER DEFAULT 1,
    exp      INTEGER DEFAULT 1,
    personality TEXT,           -- JSON
    updated  DATETIME DEFAULT CURRENT_TIMESTAMP
);
